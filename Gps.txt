    <h2>Recent Log</h2>
    <div id="historyList"></div>
    <button id="downloadBtn">📥 Export CSV</button>
</div>

<script>
    // IMPORTANT: Replace this with your Web App URL (ending in /exec), not the spreadsheet link
    const WEB_APP_URL = 'https://docs.google.com/spreadsheets/d/1HjHNF7dJCxT4oPUe3god0E8fJHr1LMyGORJ0sVFIruM/edit?gid=0#gid=0'; 
    const form = document.getElementById('dataForm');
    const historyList = document.getElementById('historyList');

    let logs = JSON.parse(localStorage.getItem('prod_logs')) || [];
    renderLogs();

    form.addEventListener('submit', function(e) {
        e.preventDefault();
        const statusMsg = document.getElementById('status');
        statusMsg.innerText = "📍 Requesting GPS...";

        // Request Geolocation
        if (navigator.geolocation) {
            navigator.geolocation.getCurrentPosition(
                (position) => {
                    document.getElementById('latField').value = position.coords.latitude;
                    document.getElementById('longField').value = position.coords.longitude;
                    sendData();
                },
                (error) => {
                    statusMsg.innerText = "⚠️ GPS Denied. Saving without location...";
                    sendData(); // Still send data even if GPS is denied
                }
            );
        } else {
            statusMsg.innerText = "❌ GPS not supported. Saving...";
            sendData();
        }
    });

    function sendData() {
        const entry = Object.fromEntries(new FormData(form));
        document.getElementById('status').innerText = "⏳ Saving to Sheet...";
        
        fetch(WEB_APP_URL + "?" + new URLSearchParams(entry), { method: "POST", mode: "no-cors" })
        .then(() => {
            logs.unshift(entry);
            if(logs.length > 5) logs.pop();
            localStorage.setItem('prod_logs', JSON.stringify(logs));
            renderLogs();
            form.reset();
            document.getElementById('status').innerText = "✅ Saved!";
            setTimeout(() => document.getElementById('status').innerText = "", 3000);
        })
        .catch(() => {
            document.getElementById('status').innerText = "❌ Connection Error";
        });
    }

    function deleteLog(index) {
        if(confirm("Delete this entry from recent history?")) {
            logs.splice(index, 1);
            localStorage.setItem('prod_logs', JSON.stringify(logs));
            renderLogs();
        }
    }

    function renderLogs() {
        historyList.innerHTML = logs.length ? logs.map((log, index) => `
            <div class="history-item">
                <div>
                    <strong>${log.Input}</strong> → ${log.Output}<br>
                    <small>Qty: ${log.Quantity} | ${log.Time}</small>
                </div>
                <button class="delete-btn" onclick="deleteLog(${index})">Delete</button>
            </div>`).join('') : "<p style='text-align:center;color:#94a3b8'>No entries</p>";
    }

    document.getElementById('downloadBtn').addEventListener('click', () => {
        let csv = "data:text/csv;charset=utf-8,Input,Output,Time,Quantity,Lat,Long\n" + 
            logs.map(l => `${l.Input},${l.Output},${l.Time},${l.Quantity},${l.Lat||''},${l.Long||''}`).join("\n");
        window.open(encodeURI(csv));
    });

    document.getElementById('clearBtn').addEventListener('click', () => form.reset());
</script>
